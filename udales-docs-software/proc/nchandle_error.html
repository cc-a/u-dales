  subroutine nchandle_error(status)
    use netcdf
    implicit none

    integer, intent(in) :: status

    if(status /= nf90_noerr) then
      write(0, *) trim(nf90_strerror(status))
      write(0, *) 'ERROR: status',  status
      stop 1
    end if

  end subroutine nchandle_error
