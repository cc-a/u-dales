   subroutine yso_Neumann

     use modglobal, only : je, jhc, rk3step, dt, dyi, nsv
     use modfields, only : sv0, svm

     real rk3coef
     integer n, m

     rk3coef = dt/(4.-dble(rk3step))

     do n = 1, nsv
       do m = 1, jhc
         sv0(:, je + m, :, n) = sv0(:, je, :, n)
         svm(:, je + m, :, n) = svm(:, je, :, n)
       end do
     end do

   end subroutine yso_Neumann
